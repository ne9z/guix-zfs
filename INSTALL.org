For vm: set serial number for disk to =test1=.

Boot guix iso.

Switch to tty3 with =C-M-f3=.  tty1 is the TUI installer, tty2 is the
info manual.

Set root password with =passwd=.

Start ssh daemon with =herd start ssh-daemon=.

Connect from another computer.

List disks with =find /dev/disk/by-id=.

Put that disk in a variable

#+begin_src sh
  DISK=/dev/disk/by-id/virtio-test1
#+end_src


Set a mount point

#+begin_src sh
  MNT=$(mktemp -d)
#+end_src


Swap, reserve

#+begin_src sh
  SWAPSIZE=4
  RESERVE=1
#+end_src

Partition

#+begin_src sh
  partition_disk () {
      local disk="${1}"
      blkdiscard -f "${disk}" || true

      parted --script --align=optimal  "${disk}" -- \
             mklabel gpt \
             mkpart EFI 2MiB 1GiB \
             mkpart guix 1GiB -$((SWAPSIZE + RESERVE))GiB \
             mkpart swap  -$((SWAPSIZE + RESERVE))GiB -"${RESERVE}"GiB \
             mkpart BIOS 1MiB 2MiB \
             set 1 esp on \
             set 3 swap on \
             set 4 bios_grub on \
             set 4 legacy_boot on

      partprobe "${disk}"
  }

  for i in ${DISK}; do
      partition_disk "${i}"
  done
#+end_src

Format fs

#+begin_src sh
  mkfs.vfat -n EFI ${DISK}-part1
  mkfs.ext4 -L guix ${DISK}-part2
  mkswap ${DISK}-part3
#+end_src

Mount fs

#+begin_src sh
  mount ${DISK}-part2 ${MNT}
  mkdir -p ${MNT}/boot/efi
  mkdir -p ${MNT}/etc
  swapon ${DISK}-part3
#+end_src

Download config

#+begin_src sh
  wget -O ${MNT}/etc/config.scm https://raw.githubusercontent.com/ne9z/guix-zfs/main/config.scm
#+end_src

Install guix

#+begin_src sh
  guix system init ${MNT}/etc/config.scm ${MNT}
#+end_src

Unmount everything

#+begin_src sh
  umount -R ${MNT}
#+end_src

Reboot

#+begin_src sh
  shutdown
#+end_src
